import matplotlib.pyplot as plt

from rydstate.rydberg_state import RydbergStateAlkali, RydbergStateAlkalineLS

Check Rubidium with large n

For Rubidium and large quantum numbers n we expect the effective model potentials to be very similar.

[2]:

state = RydbergStateAlkali("Rb", n=40, l=0, j=0.5)

states: dict[str, RydbergStateAlkali] = {}


states["model_potential_marinescu_1993"] = RydbergStateAlkali(state.species, n=state.n, l=state.l, j=state.j)
states["model_potential_marinescu_1993"].radial.create_model(potential_type="model_potential_marinescu_1993")

states["model_potential_fei_2009"] = RydbergStateAlkali(state.species, n=state.n, l=state.l, j=state.j)
states["model_potential_fei_2009"].radial.create_model(potential_type="model_potential_fei_2009")

for label, state in states.items():
    print(f"Creating wavefunction for {label}")
    state.radial.create_wavefunction()

Creating wavefunction for model_potential_marinescu_1993
Creating wavefunction for model_potential_fei_2009

[3]:

fig, axs = plt.subplots(1, 2, figsize=(12, 6))

for ax in axs:
    linestyles = ["-", "--", "-.", ":"]
    for label, state in states.items():
        ax.plot(state.radial.grid.z_list, state.radial.wavefunction.w_list, label=label, lw=2, ls=linestyles.pop(0))

    ax.legend()
    ax.set_xlabel("$z$")
    ax.set_ylabel("$w(z)$")

axs[1].set_xlim(0, 5)

plt.show()
